## re-enter root password ##

## skip the validate password plugin ##

## skip change root password ##

## remove anonymous users ##

send "y\r"

## disable remote connections ##

send "y\r"

## remote test database ##

send "y\r"

## reload privileges ##

send "y\r"

## end of script ##
EOF

## run expect script ##
