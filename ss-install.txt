##################
#### ss-check ####
##################

## delete tmp files ##

# download latest versions ##
cd /tmp/
wget --no-cache http://mirrors.slickstack.io/ss-check.txt

## rename files ##
mv ss-check.txt ss-check

## copy files to their destinations ##
cp -R -f -d --no-preserve=mode,ownership /tmp/ss-check /var/www/ss-check

## delete tmp files ##
rm -R -f /tmp/ss-check*

## reset permissions ##
chown root:root /var/www/ss-check
chmod 755 /var/www/ss-check

## call scripts ##
source /var/www/ss-check
source /var/www/ss-perms

#### repos ####
###############

## fix dpkg ##

## update repos ##
DEBIAN_FRONTEND=noninteractive DEBIAN_PRIORITY=critical PATH='/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin' /usr/bin/apt-get -q --yes -o Dpkg::Options::="--force-confdef" -o Dpkg::Options::="--force-confold" update

## install linux utilities ##
DEBIAN_FRONTEND=noninteractive DEBIAN_PRIORITY=critical PATH='/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin' /usr/bin/apt-get -q --yes -o Dpkg::Options::="--force-confdef" -o Dpkg::Options::="--force-confold" install zip unzip

## install git ##
DEBIAN_FRONTEND=noninteractive DEBIAN_PRIORITY=critical PATH='/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin' /usr/bin/apt-get -q --yes -o Dpkg::Options::="--force-confdef" -o Dpkg::Options::="--force-confold" install git

##########################
#### sudoers (visudo) ####
##########################

## delete tmp files ##
rm -R -f /tmp/sudoers.txt sudoers

## download latest versions ##
cd /tmp/
wget --no-cache http://mirrors.slickstack.io/ubuntu/sudoers.txt

## replace variables ##
sed -i "s/@SUDO/${sudo}/g" /tmp/sudoers.txt

## rename files ##
mv sudoers.txt sudoers

## copy files to their destinations ##
cp -R -f -d --no-preserve=mode,ownership /tmp/sudoers /etc/sudoers

## delete tmp files ##
rm -R -f /tmp/sudoers.txt sudoers

## reset permissions ##
chown root:root /etc/sudoers
chmod 440 /etc/sudoers

##################
#### ssh/sftp ####
##################

## delete tmp files ##
rm -R -f /tmp/sshd-config.txt sshd_config

## download latest versions ##
cd /tmp/
wget --no-cache http://mirrors.slickstack.io/ubuntu/sshd-config.txt

## replace variables ##
sed -i "s/@SUDO/${sudo}/g" /tmp/sshd-config.txt
sed -i "s/@USER/${user}/g" /tmp/sshd-config.txt

## rename files ##
mv sshd-config.txt sshd_config

## overwrite all files to their various destinations ##
cp -R -f --no-preserve=mode,ownership /tmp/sshd_config /etc/ssh/sshd_config

## delete tmp files ##
rm -R -f /tmp/sshd-config.txt sshd_config

## reset permissions ##
chown root:root /etc/ssh/sshd_config

## restart ssh ##
/etc/init.d/ssh restart

##################
#### timezone ####
##################

### wp-cli ####
###############

## download latest versions ##

## rename files and copy to their destinations ##
mv wp-cli.phar /usr/local/bin/wp

## rename files ##

## copy files to destinations ##

## reset permissions ##

#### ssl ####
#############

## generate ssl certificate ##
openssl req -new -x509 -nodes -days 730 -newkey rsa:2048 -keyout /etc/ssl/nginx.key -out /etc/ssl/nginx.crt -subj "/C=${country}/ST=${state}/L=${city}/O=${company}/OU=${dept}/CN=${domain}"

## reset permissions ##
chown root:root /etc/ssl/nginx.key
chown root:root /etc/ssl/nginx.crt

###############
#### nginx ####
###############

## update repos ##
DEBIAN_FRONTEND=noninteractive DEBIAN_PRIORITY=critical PATH='/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin' /usr/bin/apt-get -q --yes -o Dpkg::Options::="--force-confdef" -o Dpkg::Options::="--force-confold" update

## update repos ##
DEBIAN_FRONTEND=noninteractive DEBIAN_PRIORITY=critical PATH='/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin' /usr/bin/apt-get -q --yes -o Dpkg::Options::="--force-confdef" -o Dpkg::Options::="--force-confold" update

#### nginx config ####
######################

## delete tmp files ##

## download latest versions ##
cd /tmp/
wget --no-cache http://mirrors.slickstack.io/nginx/nginx-conf.txt

## rename files ##
mv nginx-conf.txt nginx.conf

## copy files to destinations ##
cp -R -f --no-preserve=mode,ownership /tmp/nginx.conf /etc/nginx/nginx.conf

## delete tmp files ##
rm -R -f /tmp/nginx-conf.txt /tmp/nginx.conf

## reset permissions ##
chown root:root /etc/nginx/nginx.conf

######################
#### nginx server ####
######################

## delete tmp files ##
rm -R -f /tmp/default.txt /tmp/default /tmp/server-block.txt /tmp/server-block /tmp/server-block-ssl.txt /tmp/server-block-ssl

## download latest versions ##
cd /tmp/
if [[ "$ssl" == "yes" ]]
then wget --no-cache http://mirrors.slickstack.io/nginx/server-block-ssl.txt
mv /tmp/server-block-ssl.txt /tmp/default.txt
else wget --no-cache http://mirrors.slickstack.io/nginx/server-block.txt
mv /tmp/server-block.txt /tmp/default.txt
fi

## replace @DOMAIN placeholder with slickstack variable ##
sed -i "s/@DOMAIN/${domain}/g" /tmp/default.txt

## rename files ##
mv /tmp/default.txt /tmp/default

## copy files to destinations ##
cp -R -f --no-preserve=mode,ownership /tmp/default /etc/nginx/sites-available/default

## delete tmp files ##
rm -R -f /tmp/default.txt /tmp/default /tmp/server-block.txt /tmp/server-block /tmp/server-block-ssl.txt /tmp/server-block-ssl

## reset permissions ##
chown root:root /etc/nginx/sites-available/default

## restart nginx ##
/etc/init.d/nginx restart

##############
#### php #####
##############

## install php and extensions ##
DEBIAN_FRONTEND=noninteractive DEBIAN_PRIORITY=critical PATH='/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin' /usr/bin/apt-get -q --yes -o Dpkg::Options::="--force-confdef" -o Dpkg::Options::="--force-confold" install php7.0 php7.0-fpm php7.0-mysql php7.0-curl php7.0-zip php7.0-gd php7.0-mcrypt php7.0-mbstring php7.0-bcmath php7.0-xml php7.0-json php7.0-soap

## delete tmp files ##
rm -R -f /tmp/php-ini.txt* /tmp/php.ini*
rm -R -f /tmp/php-fpm-conf.txt* /tmp/php-fpm.conf*
rm -R -f /tmp/www-conf.txt* /tmp/www.conf*

## download latest versions ##
cd /tmp/
wget --no-cache http://mirrors.slickstack.io/php/php-ini.txt
wget --no-cache http://mirrors.slickstack.io/php/php-fpm-conf.txt
wget --no-cache http://mirrors.slickstack.io/php/www-conf.txt

## rename files ##
mv php-ini.txt php.ini
mv php-fpm-conf.txt php-fpm.conf
mv www-conf.txt www.conf

## copy files to their destinations ##
cp -R -f --no-preserve=mode,ownership /tmp/php.ini /etc/php/7.0/fpm/php.ini
cp -R -f --no-preserve=mode,ownership /tmp/php.ini /etc/php/7.0/cli/php.ini
cp -R -f --no-preserve=mode,ownership /tmp/php-fpm.conf /etc/php/7.0/fpm/php-fpm.conf
cp -R -f --no-preserve=mode,ownership /tmp/www.conf /etc/php/7.0/fpm/pool.d/www.conf

## delete tmp files ##
rm -R -f /tmp/php-ini.txt* /tmp/php.ini*
rm -R -f /tmp/php-fpm-conf.txt* /tmp/php-fpm.conf*
rm -R -f /tmp/www-conf.txt* /tmp/www.conf*

## reset permissions ##
chown root:root /etc/php/7.0/cli/php.ini
chown root:root /etc/php/7.0/fpm/php.ini
chown root:root /etc/php/7.0/fpm/php-fpm.conf
chown root:root /etc/php/7.0/fpm/pool.d/www.conf

## restart php ##
/etc/init.d/php7.0-fpm restart
